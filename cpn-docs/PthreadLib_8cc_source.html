108 void *Pthread::Join(void) {
109  void *result = 0;
110  {
111  PthreadMutexProtected p(mutex);
112  if (inJoin || state == joined) {
113  while (state != joined) {
114  cond.Wait(mutex);
115  }
116  return returnResult;
117  }
118  inJoin = true;
119  }
120  TrapError( pthread_join(theThread, &result) );
121  {
122  PthreadMutexProtected p(mutex);
123  returnResult = result;
124  state = joined;
125  cond.Broadcast();
126  return returnResult;

131 void* Pthread::PthreadEntryPoint(void* arg)
132 //-----------------------------------------------------------------------------
133 {
134  void* result = 0;
135  Pthread* ptr = (Pthread*) arg;
136  {
137  PthreadMutexProtected p(ptr->mutex);
138  while (ptr->state == created) {
139  ptr->cond.Wait(ptr->mutex);
