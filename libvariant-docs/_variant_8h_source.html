521  template<typename T>
522  std::complex<T> VariantCaster< std::complex<T> >::Cast(const Variant &v) {
523  if (v.IsNull()) {
524  return std::complex<T>();
525  } else if (v.IsNumber()) {
526  return std::complex<T>(v.As<T>(), T());
527  } else if (v.IsMap()) {
528  T r = variant_cast<T>(v.Get("real", T()));
529  T c = variant_cast<T>(v.Get("imag", T()));
530  return std::complex<T>(r, c);
531  } else if (v.IsList()) {
532  if (v.Size() > 1) {
533  return std::complex<T>(v[0].As<T>(), v[1].As<T>());
534  } else if (v.Size() > 0) {
535  return std::complex<T>(v[0].As<T>(), T());
536  } else {
537  return std::complex<T>();
538  }
539  }
540  throw UnableToConvertError(v.GetType(), "std::complex<T>");
541  }
