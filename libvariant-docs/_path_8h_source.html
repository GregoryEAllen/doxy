29  // A simple xpath like path functionality for Variant
30  //
31  // Map keys are specified using a "/" then the key.
32  // List indeces are specified using [index]
33  // A "/" at the end of the path is ignored.
34  // Characters "\/[]" must all be escaped with "\"
35  //
36  // For example:
37  // {
38  // "key1": [ 1, 2, 3, { "blork": true } ],
39  // "key2": {
40  // "subkey": [ "a", { "furple": 1 }, 2 ],
41  // "sub/key2": true
42  // }
43  // }
44  //
45  // Then all paths enumerated are:
46  // Path Value
47  // "/" Identity
48  // "/key1" [ 1, 2, 3, { "blork": true } ],
49  // "/key1[0]" 1
50  // "/key1[1]" 2
51  // "/key1[2]" 3
52  // "/key1[3]" { "blork": true }
53  // "/key1[3]/blork" true
54  // "/key2" { "subkey": [ "a", { "furple": 1 }, 2 ], "sub/key2": true }
55  // "/key2/subkey" [ "a", { "furple": 1 }, 2 ]
56  // "/key2/subkey[0]" "a"
57  // "/key2/subkey[1]" { "furple": 1 }
58  // "/key2/subkey[1]/furple" 1
59  // "/key2/subkey[2]" 2
60  // "/key2/sub\/key2" true
61  //
62  // For all these functions PathList must act like a vector<Variant>
63  // It may be of some other type (like deque<Variant>)

65  class PathElement {

72  bool IsNumber() const { return type == INDEX; }

75  return index;

77  bool IsString() const { return type == KEY; }

81  }

86  };
